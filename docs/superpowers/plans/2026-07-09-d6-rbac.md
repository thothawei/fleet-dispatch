# D6 RBAC 多角色 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 後台使用者分三層權限（viewer/dispatcher/superadmin），含後端 DB 查詢式強制層、帳號管理 API 與前端守衛。

**Architecture:** JWT 仍簽 `role="admin"`；`AdminAuth` 驗完 token 後查 DB 取該 admin 的 `role`+`is_active`（達成停用即時生效），放進 context；`RequireAdminRole(min)` 依 rank 分級路由。superadmin 可經 `/api/admin/admins` 管理帳號（防鎖死交易）。前端以 `GET /api/admin/me` 取權威角色，做選單/路由守衛與寫入控制降級。

**Tech Stack:** 後端 Go + Gin + GORM + PostGIS + testcontainers（整合測試需 Docker）。前端 React + TS + Vite + Ant Design + TanStack Query + Vitest。

## Global Constraints

- 規格來源：`docs/superpowers/specs/2026-07-09-d6-rbac-design.md`（同 repo）。
- 角色白名單固定三值：`viewer` / `dispatcher` / `superadmin`；rank `viewer=1 < dispatcher=2 < superadmin=3`，高階含低階全部權限。
- migration 向後相容：`admins.role` 預設 `superadmin`、`is_active` 預設 `true`——既有 `admin/admin` 維持全權。
- JWT 簽發格式不變（`auth.GenerateToken("admin", admins.id, ...)`），不得把 sub-role 塞進 token。
- 帳號不提供硬刪除；「移除」＝ `is_active=false`。
- 新建/改密碼一律走 `golang.org/x/crypto/bcrypt`（沿用既有 `AdminRegistry` 模式），不得自刻雜湊。
- fleet 三 repo 慣例：直接在 `main` 開發，commit 後直接 push。
- 後端 UI 字串、log、error 用繁體中文；識別字（函式/欄位/檔名）保留英文。

// internal/auth/adminrole.go
package auth

// AdminRole 後台角色（rank 由小到大，高階含低階全部權限）
type AdminRole string

const (
	RoleViewer     AdminRole = "viewer"
	RoleDispatcher AdminRole = "dispatcher"
	RoleSuperadmin AdminRole = "superadmin"
)

var adminRoleRank = map[AdminRole]int{
	RoleViewer:     1,
	RoleDispatcher: 2,
	RoleSuperadmin: 3,
}

// ParseAdminRole 驗證並轉型；非白名單值回 ok=false
func ParseAdminRole(s string) (AdminRole, bool) {
	r := AdminRole(s)
	if _, ok := adminRoleRank[r]; ok {
		return r, true
	}
	return "", false
}

// Rank 角色等級；未知角色回 0
func (r AdminRole) Rank() int { return adminRoleRank[r] }

// AtLeast 是否具備至少 min 的權限
func (r AdminRole) AtLeast(min AdminRole) bool { return r.Rank() >= min.Rank() && r.Rank() > 0 }
```

func (r *AdminRepository) FindByID(id int64) (*model.Admin, error) {
	var a model.Admin
	if err := r.db.First(&a, id).Error; err != nil {
		return nil, err
	}
	return &a, nil
}

func (r *AdminRepository) ListAll() ([]model.Admin, error) {
	var list []model.Admin
	err := r.db.Order("id asc").Find(&list).Error
	return list, err
}

func (r *AdminRepository) UpdateAdmin(a *model.Admin) error {
	return r.db.Model(a).Select("Role", "PasswordHash", "IsActive", "UpdatedAt").Updates(a).Error
}

// CountActiveSuperadmins 可帶交易 handle；tx 為 nil 時用預設 db
func (r *AdminRepository) CountActiveSuperadmins(tx *gorm.DB) (int64, error) {
	db := r.db
	if tx != nil {
		db = tx
	}
	var n int64
	err := db.Model(&model.Admin{}).
		Where("role = ? AND is_active = ?", "superadmin", true).Count(&n).Error
	return n, err
}

// Tx 在交易內執行 fn
func (r *AdminRepository) Tx(fn func(*gorm.DB) error) error {
	return r.db.Transaction(fn)
}

// AdminLookup 依 admin id 取回角色與啟用狀態（由 repository 注入）
type AdminLookup func(id int64) (role string, active bool, err error)

// CtxAdminRole 存放經 DB 查回的 admin 角色
const CtxAdminRole = "admin_role"

// AdminAuth 驗 Bearer JWT（role=admin）後查 DB 取角色/啟用狀態；停用回 403、查無回 401
func AdminAuth(secret string, lookup AdminLookup) gin.HandlerFunc {
	return func(c *gin.Context) {
		header := c.GetHeader("Authorization")
		if !strings.HasPrefix(header, "Bearer ") {
			c.AbortWithStatusJSON(http.StatusUnauthorized, gin.H{"error": "缺少或格式錯誤的授權標頭"})
			return
		}
		role, id, err := auth.ParseToken(strings.TrimPrefix(header, "Bearer "), secret)
		if err != nil || role != "admin" {
			c.AbortWithStatusJSON(http.StatusUnauthorized, gin.H{"error": "token 無效或已過期"})
			return
		}
		adminRole, active, err := lookup(id)
		if err != nil {
			c.AbortWithStatusJSON(http.StatusUnauthorized, gin.H{"error": "帳號不存在"})
			return
		}
		if !active {
			c.AbortWithStatusJSON(http.StatusForbidden, gin.H{"error": "帳號已停用"})
			return
		}
		c.Set(CtxAdminID, id)
		c.Set(CtxAdminRole, adminRole)
		c.Next()
	}
}

// AdminRoleFromCtx 取出 AdminAuth 放入的角色
func AdminRoleFromCtx(c *gin.Context) string {
	if v, ok := c.Get(CtxAdminRole); ok {
		if s, ok := v.(string); ok {
			return s
		}
	}
	return ""
}

// RequireAdminRole 要求 context 內角色 rank >= min，否則 403
func RequireAdminRole(min auth.AdminRole) gin.HandlerFunc {
	return func(c *gin.Context) {
		role, _ := auth.ParseAdminRole(AdminRoleFromCtx(c))
		if !role.AtLeast(min) {
			c.AbortWithStatusJSON(http.StatusForbidden, gin.H{"error": "權限不足"})
			return
		}
		c.Next()
	}
}
```

package service

import (
	"errors"
	"time"

	"golang.org/x/crypto/bcrypt"
	"gorm.io/gorm"

	"line-fleet-dispatch/internal/auth"
	"line-fleet-dispatch/internal/model"
	"line-fleet-dispatch/internal/repository"
)

var (
	ErrBadRole        = errors.New("角色無效")
	ErrSelfLockout    = errors.New("不可對自己執行此操作")
	ErrLastSuperadmin = errors.New("不可移除最後一個 superadmin")
)

type AdminUsers struct {
	repo *repository.AdminRepository
}

func NewAdminUsers(repo *repository.AdminRepository) *AdminUsers { return &AdminUsers{repo: repo} }

func (s *AdminUsers) List() ([]model.Admin, error) { return s.repo.ListAll() }

func (s *AdminUsers) Create(username, password, role string) (*model.Admin, error) {
	if _, ok := auth.ParseAdminRole(role); !ok {
		return nil, ErrBadRole
	}
	hash, err := bcrypt.GenerateFromPassword([]byte(password), bcrypt.DefaultCost)
	if err != nil {
		return nil, err
	}
	now := time.Now()
	a := &model.Admin{Username: username, PasswordHash: string(hash), Role: role,
		IsActive: true, Name: username, CreatedAt: now, UpdatedAt: now}
	if err := s.repo.Create(a); err != nil {
		return nil, err
	}
	return a, nil
}

// Update 改角色／重設密碼／啟停；防自我鎖死與移除最後一個 superadmin（交易內檢查）
func (s *AdminUsers) Update(actorID, targetID int64, newRole, newPassword *string, active *bool) error {
	if newRole != nil {
		if _, ok := auth.ParseAdminRole(*newRole); !ok {
			return ErrBadRole
		}
	}
	// 對自己的降級/停用一律擋（避免把自己鎖在外面）
	if actorID == targetID {
		if (active != nil && !*active) || (newRole != nil && *newRole != string(auth.RoleSuperadmin)) {
			return ErrSelfLockout
		}
	}
	return s.repo.Tx(func(tx *gorm.DB) error {
		var target model.Admin
		if err := tx.First(&target, targetID).Error; err != nil {
			return err
		}
		demoting := newRole != nil && *newRole != string(auth.RoleSuperadmin) && target.Role == string(auth.RoleSuperadmin)
		deactivating := active != nil && !*active && target.IsActive
		if (demoting || deactivating) && target.Role == string(auth.RoleSuperadmin) {
			n, err := s.repo.CountActiveSuperadmins(tx)
			if err != nil {
				return err
			}
			if n <= 1 {
				return ErrLastSuperadmin
			}
		}
		if newRole != nil {
			target.Role = *newRole
		}
		if active != nil {
			target.IsActive = *active
		}
		if newPassword != nil {
			hash, err := bcrypt.GenerateFromPassword([]byte(*newPassword), bcrypt.DefaultCost)
			if err != nil {
				return err
			}
			target.PasswordHash = string(hash)
		}
		target.UpdatedAt = time.Now()
		return tx.Model(&target).
			Select("Role", "IsActive", "PasswordHash", "UpdatedAt").Updates(&target).Error
	})
}

import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { Table, Button, Modal, Form, Input, Select, Switch, message, Tag } from 'antd';
import { useState } from 'react';
import { listAdmins, createAdmin, updateAdmin, type AdminUser } from '../api/admin';

const ROLE_OPTS = [
  { value: 'viewer', label: '檢視者' },
  { value: 'dispatcher', label: '派單員' },
  { value: 'superadmin', label: '超級管理員' },
];

export default function UsersPage() {
  const qc = useQueryClient();
  const [open, setOpen] = useState(false);
  const [form] = Form.useForm();
  const { data = [] } = useQuery({ queryKey: ['admins'], queryFn: listAdmins });
  const refresh = () => qc.invalidateQueries({ queryKey: ['admins'] });
  const onErr = (e: any) => message.error(e.response?.data?.error ?? '操作失敗');

  const createMut = useMutation({
    mutationFn: createAdmin,
    onSuccess: () => { setOpen(false); form.resetFields(); refresh(); message.success('已建立'); },
    onError: onErr,
  });
  const updateMut = useMutation({
    mutationFn: (v: { id: number; patch: any }) => updateAdmin(v.id, v.patch),
    onSuccess: () => { refresh(); }, onError: onErr,
  });

  const columns = [
    { title: '帳號', dataIndex: 'username' },
    {
      title: '角色', dataIndex: 'role',
      render: (r: string, row: AdminUser) => (
        <Select size="small" value={r} options={ROLE_OPTS} style={{ width: 130 }}
          onChange={(role) => updateMut.mutate({ id: row.id, patch: { role } })} />
      ),
    },
    {
      title: '啟用', dataIndex: 'is_active',
      render: (v: boolean, row: AdminUser) => (
        <Switch checked={v} onChange={(is_active) => updateMut.mutate({ id: row.id, patch: { is_active } })} />
      ),
    },
  ];

  return (
    <>
      <Button type="primary" style={{ marginBottom: 16 }} onClick={() => setOpen(true)}>新增帳號</Button>
      <Table rowKey="id" dataSource={data} columns={columns as any} />
      <Modal title="新增後台帳號" open={open} onOk={() => form.submit()} onCancel={() => setOpen(false)}>
        <Form form={form} layout="vertical" onFinish={(v) => createMut.mutate(v)}>
          <Form.Item name="username" label="帳號" rules={[{ required: true }]}><Input /></Form.Item>
          <Form.Item name="password" label="密碼" rules={[{ required: true, min: 6 }]}><Input.Password /></Form.Item>
          <Form.Item name="role" label="角色" rules={[{ required: true }]}><Select options={ROLE_OPTS} /></Form.Item>
        </Form>
      </Modal>
    </>
  );
}
